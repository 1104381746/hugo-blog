{{- define "content" -}} {{- $type := .Type -}} {{- $hierarchy := hugo.Data.categories -}} {{- /* Collect all child category names for filtering */ -}} {{- $allChildren := slice -}} {{- range $_, $children := $hierarchy -}} {{- range $c := $children -}} {{- $allChildren = $allChildren | append $c -}} {{- end -}} {{- end -}}

{{- .Params.Title | default (T "categories") | default "categories" | dict "Some" | T "allSome" -}}

{{- /* Render standalone categories (neither parent nor child) */ -}} {{- range .Data.Terms.Alphabetical -}} {{- $termName := .Term -}} {{- $isParent := index $hierarchy $termName -}} {{- $isChild := in $allChildren $termName -}} {{- if and (not $isParent) (not $isChild) -}} {{- with $.Site.GetPage "taxonomy" (printf "%v/%v" $type $termName) -}}

 {{ .Page.Title }}

{{- $termPages := index $.Site.Taxonomies.categories $termName -}} {{- range first 5 $termPages.Pages -}}
{{ .Title | emojify }}
{{- end -}} {{- if gt (len $termPages.Pages) 5 -}} {{ T "more" }} >> {{- end -}}
{{- end -}} {{- end -}} {{- end -}} {{- /* Render parent categories with nested children */ -}} {{- range $parent, $children := $hierarchy -}} {{- $parentPage := $.Site.GetPage "taxonomy" (printf "%v/%v" $type $parent) -}}

{{- if $parentPage -}}  {{ $parent }} {{- else -}}  {{ $parent }} {{- end -}}

{{- range $child := $children -}} {{- $childPage := $.Site.GetPage "taxonomy" (printf "%v/%v" $type $child) -}} {{- if $childPage -}}

 {{ $child }}

{{- $childTerm := index $.Site.Taxonomies.categories $child -}} {{- range first 5 $childTerm.Pages -}}
{{ .Title | emojify }}
{{- end -}} {{- if gt (len $childTerm.Pages) 5 -}} {{ T "more" }} >> {{- end -}}
{{- end -}} {{- end -}}
{{- end -}}
{{- end -}}
